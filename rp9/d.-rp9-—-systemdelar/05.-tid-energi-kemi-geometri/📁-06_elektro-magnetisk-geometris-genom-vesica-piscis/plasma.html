# -*- coding: utf-8 -*-
# plasma_node_flow_map.py ‚Äî Fraktal nodkarta + urladdningsstig i ett plasma-inspirerat n√§t.
# Visualiserar hur "str√∂mmen" r√∂r sig nod-till-nod l√§ngs rakaste v√§g i energitermer.
#
# Matematik i modellen:
# - Noder placeras av primtal: r_n = sqrt(p_n), theta_n = p_n * pi / 180, x = r cos theta, y = r sin theta.
# - Grafvikter: w_ij = d_ij^(-gamma) med d_ij = euklidiskt avst√•nd.
# - Dirichletvillkor: Q(source)=1, Q(sink)=0. Inre noder l√∂ser diskret Laplace: sum_j w_ij (Q_i - Q_j) = 0.
# - Kantstr√∂mmar: I_ij = w_ij (Q_i - Q_j).
# - Urladdningsstig: v√§lj granne som maximerar (Q_i - Q_j)/d_ij^alpha under Q_j < Q_i.
#
# Viktig modellnot:
# - phi = 1.5 anv√§nds som exakt harmonisk kalibreringskonstant enligt din modell (INTE approximation).
#   Som default s√§tter vi alpha = phi = 1.5 f√∂r stig-urvalet; detta kan √§ndras via flagga.

def primes_first_n(n):
    # Enkel generator av de n f√∂rsta primtalen.
    if n <= 0:
        return []
    res = []
    cand = 2
    while len(res) < n:
        is_p = True
        r = int(math.sqrt(cand))
        for p in res:
            if p > r:
                break
            if cand % p == 0:
                is_p = False
                break
        if is_p:
            res.append(cand)
        cand += 1 if cand == 2 else 2
    return res

def polar_prime_positions(n):
    # Placera n noder via f√∂rsta n primtal: r = sqrt(p), theta = p*pi/180.
    ps = primes_first_n(n)
    coords = []
    for p in ps:
        r = math.sqrt(p)
        theta = p * math.pi / 180.0
        x = r * math.cos(theta)
        y = r * math.sin(theta)
        coords.append((x, y, r, theta, p))
    return coords

def euclid(a, b):
    dx = a[0] - b[0]
    dy = a[1] - b[1]
    return math.hypot(dx, dy)

def build_knn_edges(coords, k):
    # Skapar KNN-kanter f√∂r varje nod. Returnerar kantlista med (i, j, d).
    n = len(coords)
    edges = set()
    for i in range(n):
        dlist = []
        for j in range(n):
            if i == j:
                continue
            d = euclid(coords[i], coords[j])
            dlist.append((d, j))
        dlist.sort(key=lambda t: t[0])
        for t in dlist[:k]:
            j = t[1]
            a = i if i < j else j
            b = j if i < j else i
            edges.add((a, b))
    out = []
    for (i, j) in edges:
        d = euclid(coords[i], coords[j])
        out.append((i, j, d))
    return out

def adjacency_with_weights(n, edges, gamma):
    # Bygg adjacency med vikter w_ij = d^{-gamma}.
    adj = [[] for _ in range(n)]
    for (i, j, d) in edges:
        w = d ** (-gamma) if d > 0.0 else 1e9
        adj[i].append((j, d, w))
        adj[j].append((i, d, w))
    return adj

def solve_dirichlet(adj, source, sink, tol=1e-6, max_iter=20000):
    # L√∂s Q f√∂r inre noder med Gauss-Seidel: Q_i = sum_j w_ij Q_j / sum_j w_ij f√∂r i ‚â† {source, sink}.
    n = len(adj)
    Q = [0.0] * n
    Q[source] = 1.0
    Q[sink] = 0.0
    fixed = {source, sink}
    # Initiera med viktat medel av grannar f√∂r snabbare start.
    for i in range(n):
        if i in fixed:
            continue
        wsum = 0.0
        acc = 0.0
        for (j, d, w) in adj[i]:
            acc += w * Q[j]
            wsum += w
        Q[i] = acc / wsum if wsum > 0 else 0.0
    # Gauss-Seidel
    for it in range(max_iter):
        err = 0.0
        for i in range(n):
            if i in fixed:
                continue
            wsum = 0.0
            acc = 0.0
            ai = adj[i]
            for (j, d, w) in ai:
                acc += w * Q[j]
                wsum += w
            newq = acc / wsum if wsum > 0 else Q[i]
            err = max(err, abs(newq - Q[i]))
            Q[i] = newq
        if err < tol:
            break
    return Q

def currents_on_edges(edges, Q, gamma):
    # I_ij = w_ij (Q_i - Q_j), w_ij = d^{-gamma}
    I = {}
    for (i, j, d) in edges:
        w = d ** (-gamma) if d > 0.0 else 1e9
        val = w * (Q[i] - Q[j])
        I[(i, j)] = val
        I[(j, i)] = -val
    return I

def greedy_discharge_path(adj, Q, source, sink, alpha, max_steps=100000):
    # Stegvis v√§lj granne j med Q_j < Q_i som maximerar (Q_i - Q_j)/d^alpha. Fallback: min Q_j.
    path = [source]
    visited = set([source])
    cur = source
    steps = 0
    while cur != sink and steps < max_steps:
        steps += 1
        candidates = []
        fallback = []
        for (j, d, w) in adj[cur]:
            if j == cur:
                continue
            qdiff = Q[cur] - Q[j]
            score = qdiff / (d ** alpha) if d > 0.0 else qdiff * 1e9
            if Q[j] < Q[cur]:
                candidates.append((score, j, d, qdiff))
            else:
                fallback.append((Q[j], j, d, qdiff))
        if candidates:
            candidates.sort(key=lambda t: (-t[0], t[3], -1.0 / max(t[2], 1e-12)))
            nxt = candidates[0][1]
        else:
            if not fallback:
                break
            fallback.sort(key=lambda t: (t[0], t[2]))
            nxt = fallback[0][1]
        path.append(nxt)
        if nxt in visited and nxt != sink:
            # Cykel, avsluta f√∂r s√§kerhet.
            break
        visited.add(nxt)
        cur = nxt
    return path

def export_json(path, coords, edges, Q, I, source, sink, path_nodes, alpha, gamma, phi_exact):
    data = {}
    data["model"] = {"alpha": alpha, "gamma": gamma, "phi_exact"] = phi_exact
    nodes = []
    for idx, c in enumerate(coords):
        nodes.append({"id": idx, "x": c[0], "y": c[1], "r": c[2], "theta": c[3], "prime": c[4], "Q": Q[idx]})
    eds = []
    for (i, j, d) in edges:
        eds.append({"i": i, "j": j, "d": d, "I_ij": I.get((i, j), 0.0), "I_ji": I.get((j, i), 0.0)})
    data["nodes"] = nodes
    data["edges"] = eds
    data["source"] = int(source)
    data["sink"] = int(sink)
    data["path"] = [int(v) for v in path_nodes]
    with open(path, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)

def pick_source_sink_by_radius(coords):
    # V√§lj min r som k√§lla och max r som s√§nka.
    rvals = [(c[2], i) for i, c in enumerate(coords)]
    rvals.sort(key=lambda t: t[0])
    source = rvals[0][1]
    sink = rvals[-1][1]
    return source, sink

def main():

    parser.add_argument("--n", type=int, default=220, help="Antal noder (f√∂rsta n primtal).")
    parser.add_argument("--k", type=int, default=4, help="Antal n√§rmaste grannar per nod (KNN-kanter).")
    parser.add_argument("--alpha", type=float, default=1.5, help="Exponent i stig-urvalet: (Q_i - Q_j)/d^alpha. Default = 1.5.")
    parser.add_argument("--gamma", type=float, default=1.0, help="Vikt-exponent i w_ij = d^(-gamma).")
    parser.add_argument("--tol", type=float, default=1e-6, help="Tolerans f√∂r Gauss-Seidel.")
    parser.add_argument("--max_iter", type=int, default=20000, help="Max iterationer i Gauss-Seidel.")
    parser.add_argument("--seed", type=int, default=0, help="Slumpfr√∂ (f√∂r stabila val om lika).")
    parser.add_argument("--outdir", default="PlasmaFlow", help="Utdatakatalog.")
    parser.add_argument("--open", action="store_true", help="√ñppna HTML/SVG i webbl√§sare efter k√∂rning.")
    args = parser.parse_args()

    random.seed(args.seed)
    os.makedirs(args.outdir, exist_ok=True)
    out_html = os.path.join(args.outdir, "plasma_flow.html")
    out_svg = os.path.join(args.outdir, "plasma_flow_fallback.svg")
    out_json = os.path.join(args.outdir, "plasma_flow.json")

    # 1) Noder fr√•n primtal
    coords = polar_prime_positions(args.n)

    # 2) KNN-kanter
    edges = build_knn_edges(coords, args.k)

    # 3) Adjacency och vikter
    adj = adjacency_with_weights(len(coords), edges, args.gamma)

    # 4) K√§lla och s√§nka v√§ljs via radie
    source, sink = pick_source_sink_by_radius(coords)

    # 5) L√∂s Q via Gauss-Seidel
    Q = solve_dirichlet(adj, source, sink, tol=args.tol, max_iter=args.max_iter)

    # 6) Str√∂mmar p√• kanter
    I = currents_on_edges(edges, Q, args.gamma)

    # 7) Urladdningsstig med alpha (default alpha = phi = 1.5 exakt enligt modellen)
    phi_exact = 1.5
    alpha = args.alpha
    path_nodes = greedy_discharge_path(adj, Q, source, sink, alpha)

    # 8) Export JSON
    export_json(out_json, coords, edges, Q, I, source, sink, path_nodes, alpha, args.gamma, phi_exact)

    # 9) Visualisering
    ok = render_plotly(coords, edges, path_nodes, Q, source, sink, out_html)
    if not ok:
        render_svg(coords, edges, path_nodes, Q, source, sink, out_svg)
        target = out_svg
    else:
        target = out_html

    print("Klar. Sparade data till:", out_json)
    print("Visualisering:", target)
    if args.open:
        webbrowser.open("file://" + os.path.abspath(target).replace("\\", "/"))

if __name__ == "__main__":
    main()

def escape_html_except_math(s: str) -> str:
    # L√§mna $...$ or√∂rt s√• MathJax kan rendera; escapa √∂vrig text.
    out = []
    i = 0
    in_math = False
    buf = []
    while i < len(s):
        ch = s[i]
        if ch == "$":
            if buf:
                seg = "".join(buf)
                if in_math:
                    out.append(seg)
                else:
                    out.append(html.escape(seg, quote=True))
                buf = []
            out.append("$")
            in_math = not in_math
            i += 1
            continue
        buf.append(ch)
        i += 1
    if buf:
        seg = "".join(buf)
        if in_math:
            out.append(seg)
        else:
            out.append(html.escape(seg, quote=True))
    return "".join(out)

def replace_inline_patterns(text: str) -> str:
    # `kod` ‚Üí <code>kod</code>
    text = re.sub(r"`([^`]+)`", lambda m: "<code>" + html.escape(m.group(1), quote=True) + "</code>", text)
    # **fet**
    text = re.sub(r"\*\*([^\*]+)\*\*", r"<strong>\1</strong>", text)
    # *kursiv* (efter fet)
    text = re.sub(r"(?<!\*)\*([^\*]+)\*(?!\*)", r"<em>\1</em>", text)
    # [l√§nk](url)
    def _link(m):
        label = m.group(1)
        url = m.group(2)
        return '<a href="' + html.escape(url, quote=True) + '" target="_blank" rel="noopener noreferrer">' + label + "</a>"
    text = re.sub(r"\[([^\]]+)\]\(([^)]+)\)", _link, text)
    return text

def slugify(s: str) -> str:
    s2 = re.sub(r"[^\w\s\-]", "", s, flags=re.UNICODE)
    s2 = re.sub(r"\s+", "-", s2.strip(), flags=re.UNICODE)
    return s2.lower()

def md_to_html(md: str):
    # Returnerar (html_str, metadata_dict)
    lines = md.split("\n")
    html_lines = []
    in_code = False
    code_lang = ""
    in_ul = False
    in_ol = False
    headings = []          # [(level, text, id)]
    code_blocks = []       # [(lang, line_count)]
    inline_math_all = []   # lista med alla $...$ segment (r√• text)
    term_counter = Counter()

    # mycket liten stopplista (sv + en) f√∂r termfrekvens
    stop = set([
        "och","det","att","som","en","i","jag","p√•","√§r","f√∂r","med","du","den","har","inte","om","vi",
        "till","av","fr√•n","de","eller","s√•","att","men","s√•","kan","ska","utan","ett","han","hon","man",
        "the","and","for","with","from","that","this","are","not","you","your","our","their","have","has",
        "of","in","to","on","is","it","be","as","by","at","an","we","they"
    ])

    def close_lists():
        nonlocal in_ul, in_ol
        if in_ul:
            html_lines.append("</ul>")
            in_ul = False
        if in_ol:
            html_lines.append("</ol>")
            in_ol = False

    # hj√§lpare: extrahera inline-$...$ och returnera (renderbar text)
    def collect_inline_math(txt: str) -> str:
        buf = []
        i = 0
        in_m = False
        seg = []
        while i < len(txt):
            ch = txt[i]
            if ch == "$":
                if in_m:
                    inline_math_all.append("".join(seg))
                    seg = []
                    in_m = False
                    buf.append("$")
                    i += 1
                    continue
                else:
                    in_m = True
                    buf.append("$")
                    i += 1
                    continue
            if in_m:
                seg.append(ch)
            else:
                buf.append(ch)
            i += 1
        # obalans ignoreras
        return "".join(buf)

    for raw in lines:
        # kodblock
        if raw.strip().startswith("```"):
            if not in_code:
                close_lists()
                in_code = True
                code_lang = raw.strip()[3:].strip()
                html_lines.append('<pre class="code"><code data-lang="' + html.escape(code_lang, quote=True) + '">')
                code_blocks.append([code_lang or "", 0])
            else:
                html_lines.append("</code></pre>")
                in_code = False
                code_lang = ""
            continue

        if in_code:
            html_lines.append(html.escape(raw, quote=True))
            if code_blocks:
                code_blocks[-1][1] += 1
            continue

        # tom rad
        if raw.strip() == "":
            close_lists()
            html_lines.append("")
            continue

        # rubriker
        m = re.match(r"^(#{1,6})\s+(.*)$", raw)
        if m:
            close_lists()
            level = len(m.group(1))
            text = m.group(2).strip()
            text = collect_inline_math(text)
            text_escaped = escape_html_except_math(text)
            text_inline = replace_inline_patterns(text_escaped)
            hid = slugify(text)
            headings.append((level, text, hid))
            html_lines.append("<h" + str(level) + ' id="' + hid + '">' + text_inline + "</h" + str(level) + ">")
            # termer fr√•n rubriker v√§ger vi lite extra
            for w in re.findall(r"[A-Za-z√Ö√Ñ√ñ√•√§√∂0-9\-\_]{3,}", text.lower()):
                if w not in stop:
                    term_counter[w] += 3
            continue

        # punktlista
        if re.match(r"^\s*[\-\*]\s+.+$", raw):
            if not in_ul:
                close_lists()
                in_ul = True
                html_lines.append("<ul>")
            item = re.sub(r"^\s*[\-\*]\s+", "", raw)
            item = collect_inline_math(item)
            item = replace_inline_patterns(escape_html_except_math(item))
            html_lines.append("<li>" + item + "</li>")
            for w in re.findall(r"[A-Za-z√Ö√Ñ√ñ√•√§√∂0-9\-\_]{3,}", item.lower()):
                if w not in stop:
                    term_counter[w] += 1
            continue

        # numrerad lista
        if re.match(r"^\s*\d+\.\s+.+$", raw):
            if not in_ol:
                close_lists()
                in_ol = True
                html_lines.append("<ol>")
            item = re.sub(r"^\s*\d+\.\s+", "", raw)
            item = collect_inline_math(item)
            item = replace_inline_patterns(escape_html_except_math(item))
            html_lines.append("<li>" + item + "</li>")
            for w in re.findall(r"[A-Za-z√Ö√Ñ√ñ√•√§√∂0-9\-\_]{3,}", item.lower()):
                if w not in stop:
                    term_counter[w] += 1
            continue

        # paragraf
        raw2 = collect_inline_math(raw)
        txt = replace_inline_patterns(escape_html_except_math(raw2))
        html_lines.append("<p>" + txt + "</p>")
        for w in re.findall(r"[A-Za-z√Ö√Ñ√ñ√•√§√∂0-9\-\_]{3,}", raw2.lower()):
            if w not in stop:
                term_counter[w] += 1

    close_lists()

    meta = {
        "headings": headings,
        "code_blocks": code_blocks,
        "inline_math": inline_math_all,
        "terms": term_counter
    }
    return "\n".join(html_lines), meta

def build_head(title: str) -> str:
    parts = []
    parts.append("<meta charset=\"utf-8\">")
    parts.append("<meta name=\"viewport\" content=\"width=device-width, initial-scale=1\">")
    parts.append("<title>" + html.escape(title, quote=True) + "</title>")
    parts.append("<style>")
    parts.append(build_css())
    parts.append("</style>")
    # MathJax
    parts.append("<script>")
    parts.append("var MathJax = { tex: { inlineMath: [['$','$'], ['\\\\(','\\\\)']] }, svg: { fontCache: 'global' } };")
    parts.append("</script>")
    parts.append("<script src=\"https://cdn.jsdelivr.net/npm/mathjax@3/es5/tex-svg.js\" async></script>")
    # TOC byggare
    js = [
        "(function(){",
        "  function collect(){",
        "    var a=document.querySelector('article'); if(!a) return [];",
        "    var hs=a.querySelectorAll('h1,h2,h3');",
        "    var out=[];",
        "    for(var i=0;i<hs.length;i++){",
        "      var h=hs[i], t=h.textContent||h.innerText||'', id=h.id||'';",
        "      out.push({tag:h.tagName.toLowerCase(), text:t, id:id});",
        "    }",
        "    return out;",
        "  }",
        "  function cls(tag){ if(tag==='h1') return 'h1'; if(tag==='h2') return 'h2'; return 'h3'; }",
        "  function build(){",
        "    var items=collect();",
        "    var toc=document.querySelector('.toc'); if(!toc) return;",
        "    toc.innerHTML='';",
        "    for(var i=0;i<items.length;i++){",
        "      var it=items[i];",
        "      var a=document.createElement('a');",
        "      a.className=cls(it.tag); a.href=it.id?('#'+it.id):'#'; a.textContent=it.text;",
        "      toc.appendChild(a);",
        "    }",
        "  }",
        "  document.addEventListener('DOMContentLoaded', build);",
        "})();"
    ]
    parts.append("<script>" + "\n".join(js) + "</script>")
    return "\n".join(parts)

def build_seriesbar(series: str, part: int, prev_href: str, prev_title: str, next_href: str, next_title: str) -> str:
    if not series:
        return ""
    items = []
    items.append("<div class=\"seriesbar\">")
    crumb = series + " ¬∑ Del " + str(part)
    items.append("<div class=\"crumb\">" + html.escape(crumb, quote=True) + "</div>")
    pager = []
    pager.append("<div class=\"pager\">")
    if prev_href:
        label = "‚óÄ " + (prev_title if prev_title else "F√∂reg√•ende")
        pager.append("<a href=\"" + html.escape(prev_href, quote=True) + "\">" + html.escape(label, quote=True) + "</a>")
    if next_href:
        label2 = (next_title if next_title else "N√§sta") + " ‚ñ∂"
        pager.append("<a href=\"" + html.escape(next_href, quote=True) + "\">" + html.escape(label2, quote=True) + "</a>")
    pager.append("</div>")
    items.append("".join(pager))
    items.append("</div>")
    return "\n".join(items)

def format_terms_html(counter: Counter, topn: int = 15) -> str:
    items = counter.most_common(topn)
    if not items:
        return "<p>Inga distinkta termer hittades.</p>"
    lis = []
    for w, c in items:
        lis.append("<li><code>" + html.escape(w, quote=True) + "</code> ‚Äî " + str(c) + "</li>")
    return "<ul class=\"klist\">" + "".join(lis) + "</ul>"

def format_headings_html(heads) -> str:
    if not heads:
        return "<p>Inga rubriker hittades.</p>"
    lis = []
    for level, text, hid in heads:
        tag = "H" + str(level)
        lis.append("<li><span class=\"kbd\">" + tag + "</span> " + html.escape(text, quote=True) + "</li>")
    return "<ul class=\"klist\">" + "".join(lis) + "</ul>"

def format_math_html(math_list, maxn: int = 25) -> str:
    if not math_list:
        return "<p>Inga LaTeX-segment ($...$) hittades.</p>"
    seen = set()
    out = []
    for seg in math_list:
        if seg in seen:
            continue
        seen.add(seg)
        out.append("<li>$" + seg + "$</li>")
        if len(out) >= maxn:
            break
    return "<ul class=\"klist\">" + "".join(out) + "</ul>"

def format_codeblocks_html(blocks) -> str:
    if not blocks:
        return "<p>Inga kodblock hittades.</p>"
    lis = []
    for lang, lines in blocks:
        label = lang if lang else "text"
        lis.append("<li><span class=\"kbd\">" + html.escape(label, quote=True) + "</span> ‚Äî " + str(lines) + " rader</li>")
    return "<ul class=\"klist\">" + "".join(lis) + "</ul>"

def build_auto_afterword(meta) -> str:
    parts = []
    parts.append("<div class=\"appendix\">")
    parts.append("<h2 id=\"efterord-auto\">Efterord ‚Äî Autoanalys</h2>")
    parts.append("<p>Detta avsnitt genereras automatiskt fr√•n dokumentets inneh√•ll f√∂r att fungera som en sammanfattande appendix i slutet. Fokus ligger p√• struktur, termer, matematik och kodblock.</p>")
    parts.append("<h3>Rubrik√∂versikt (H1‚ÄìH2)</h3>")
    heads12 = [(lvl, txt, hid) for (lvl, txt, hid) in meta.get("headings", []) if lvl in (1,2)]
    parts.append(format_headings_html(heads12))
    parts.append("<h3>Termfrekvens (topp 15)</h3>")
    parts.append(format_terms_html(meta.get("terms", Counter()), 15))
    parts.append("<h3>LaTeX-segment (unika, upp till 25)</h3>")
    parts.append(format_math_html(meta.get("inline_math", []), 25))
    parts.append("<h3>Kodblock</h3>")
    parts.append(format_codeblocks_html(meta.get("code_blocks", [])))
    parts.append("</div>")
    return "\n".join(parts)

def assemble_html(content_html: str, title: str, subtitle: str, src_name: str,
                  series: str, part: int,
                  prev_href: str, prev_title: str, next_href: str, next_title: str,
                  appendix_html: str) -> str:
    head = build_head(title)
    seriesbar_html = build_seriesbar(series, part, prev_href, prev_title, next_href, next_title)
    header_block = build_header_block(title, subtitle, src_name, seriesbar_html)
    doc = []
    doc.append("<!DOCTYPE html>")
    doc.append("<html lang=\"sv\">")
    doc.append("<head>")
    doc.append(head)
    doc.append("</head>")
    doc.append("<body>")
    doc.append("<main>")
    # SIDOMENY
    doc.append("<nav>")
    doc.append("<div class=\"brand\">Inneh√•ll</div>")
    doc.append("<div class=\"sub\">Auto fr√•n rubriker (H1‚ÄìH3)</div>")
    doc.append("<div class=\"toc\"></div>")
    doc.append("</nav>")
    # ARTIKEL
    doc.append("<article>")
    doc.append(header_block)
    doc.append(content_html)
    if appendix_html:
        doc.append(appendix_html)
    doc.append("<div class=\"sig\">Rapport genererad automatiskt. Matematik renderas med MathJax ($...$).<\/div>")
    doc.append("</article>")
    doc.append("</main>")
    doc.append("</body>")
    doc.append("</html>")
    return "\n".join(doc)

def greedy_discharge_path(adj, Q, source, sink, alpha, max_steps=100000):

        # Dijkstra-liknande greedy p√• |Iuv|/d (prioritera stark och kort)
        pq = []
        import heapq
        heapq.heapify(pq)
        # (-score, node, pathlist)
        heapq.heappush(pq, (-1e9, source, [source]))
        best_path = None
        visited = set()
        while pq:
            negs, u, curp = heapq.heappop(pq)
            if (u, len(curp)) in visited:
                continue
            visited.add((u, len(curp)))
            if u == sink:
                best_path = curp
                break
            for (v, d, Iuv) in adj.get(u, []):
                if (u, v) in used:
                    continue
                score = (abs(Iuv) / max(d, 1e-12))
                heapq.heappush(pq, (-score, v, curp + [v]))
        if not best_path or len(best_path) < 2:
            break
        # Markera anv√§nda kanter
        for a, b in zip(best_path[:-1], best_path[1:]):
            used.add((a, b))
        paths.append(best_path)
    return paths

def render_svg(coords, edges, main_path, fil_paths, Q, source, sink, out_svg, width=1000, height=1000, pad=20):
    xs = [c[0] for c in coords]
    ys = [c[1] for c in coords]
    xmin, xmax = min(xs), max(xs)
    ymin, ymax = min(ys), max(ys)
    sx = (width - 2 * pad) / max(1e-9, (xmax - xmin))
    sy = (height - 2 * pad) / max(1e-9, (ymax - ymin))
    def tx(x): return pad + (x - xmin) * sx
    def ty(y): return height - (pad + (y - ymin) * sy)

    qmin = min(Q)
    qmax = max(Q)
    def qcol(q):
        t = 0.0 if qmax == qmin else (q - qmin) / (qmax - qmin)
        r = int(68 + t * (253 - 68))
        g = int(1 + t * (231 - 1))
        b = int(84 + t * (37 - 84))
        return "rgb(" + str(r) + "," + str(g) + "," + str(b) + ")"

    parts = []
    parts.append("<svg xmlns=\"http://www.w3.org/2000/svg\" width=\"" + str(width) + "\" height=\"" + str(height) + "\" viewBox=\"0 0 " + str(width) + " " + str(height) + "\">")

def export_json(path, coords, edges, Q, I, source, sink, main_path, fil_paths, alpha, gamma, kappa, phi_exact):
    data = {}
    data["model"] = {"alpha": alpha, "gamma": gamma, "kappa": kappa, "phi_exact": phi_exact}
    nodes = []
    for idx, c in enumerate(coords):
        nodes.append({"id": idx, "x": c[0], "y": c[1], "r": c[2], "theta": c[3], "prime": c[4], "Q": Q[idx]})
    eds = []
    for (i, j, d) in edges:
        eds.append({"i": i, "j": j, "d": d, "I_ij": I.get((i, j), 0.0), "I_ji": I.get((j, i), 0.0)})
    data["nodes"] = nodes
    data["edges"] = eds
    data["source"] = int(source)
    data["sink"] = int(sink)
    data["main_path"] = [int(v) for v in main_path]
    data["filaments"] = [[int(v) for v in p] for p in fil_paths]
    with open(path, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
