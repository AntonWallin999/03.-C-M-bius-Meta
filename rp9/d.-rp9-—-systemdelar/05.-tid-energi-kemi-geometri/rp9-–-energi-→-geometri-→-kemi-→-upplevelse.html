def sigmoid(x):
    return 1.0 / (1.0 + np.exp(-x))

def r2_score(y_true, y_pred):
    ss_res = np.sum((y_true - y_pred) ** 2.0)
    ss_tot = np.sum((y_true - np.mean(y_true)) ** 2.0)
    return 1.0 - (ss_res / ss_tot + 1e-15)

def soft_entropy_base2(p):
    p = np.clip(p, 1e-12, 1.0)
    return float(-np.sum(p * np.log2(p)))

def add_intercept(X):
    return np.hstack([np.ones((X.shape[0], 1)), X])

# ------------------------------
# Konstanter och niv√•er
# ------------------------------
C0  = 1.0
C1  = math.sqrt(2.0)
C2  = math.pi
PHI = 1.5            # EXAKT enligt modellen
C10 = 1.0

C = np.array([C0, C1, C2, PHI, C10], dtype=float)

# Riktad graf (kedja, bidi)
A = np.zeros((5,5), dtype=float)
for i in range(4):
    A[i, i+1] = 1.0
    A[i+1, i] = 1.0

def S_matrix(C, beta):
    S = np.zeros((5,5), dtype=float)
    for l in range(5):
        for m in range(5):
            if A[l, m] > 0.0:
                R = C[m] / C[l]
                S[l, m] = 0.5 * (1.0 + math.tanh(beta * math.log(R)))
    return S

# ------------------------------
# Bias fr√•n residual Œµ_œÄ(u)
# ------------------------------
def epsilon_pi(u):
    return math.sin(2.0 * math.pi * u) - u

def B_of_u(u, alpha):
    return sigmoid(alpha * epsilon_pi(u))

def barB_from_phase(u, alpha):
    d = int(math.floor(10.0 * u)) % 10
    p = np.ones(10, dtype=float) * (0.3 / 9.0)
    p[d] = 0.7
    mids = (np.arange(10) + 0.5) / 10.0
    Bvals = np.array([B_of_u(x, alpha) for x in mids])
    return float(np.sum(Bvals * p)), p

def simulate_time_flow(N=4000, seed=20250823, alpha=12.0, beta=1.1,
                       kappa=1.0, eta=1e-2, tau0=1.0, eps_info=1e-3,
                       omega=math.sqrt(2.0)-1.0):
    rng = np.random.default_rng(seed)
    E = np.ones(5, dtype=float) / 5.0
    S = S_matrix(C, beta)
    u = 0.12345

    Phi = np.zeros(N, dtype=float)
    Dt  = np.zeros(N, dtype=float)
    Dt_info = np.zeros(N, dtype=float)
    H = np.zeros(N, dtype=float)

    for n in range(N):
        u = (u + omega) % 1.0
        bB, p = barB_from_phase(u, alpha)
        H[n] = soft_entropy_base2(p)

        J = np.zeros((5,5), dtype=float)
        for l in range(5):
            for m in range(5):
                if A[l, m] > 0.0:
                    J[l, m] = kappa * E[l] * S[l, m] * (eta + (1.0 - eta) * bB)

        inflow  = np.sum(J, axis=0)
        outflow = np.sum(J, axis=1)
        E = E - outflow + inflow
        E = np.clip(E, 1e-12, None)
        E = E / np.sum(E)

        phi_n = float(np.sum(J))
        Phi[n] = phi_n if phi_n > 1e-15 else 1e-15

        Dt[n] = tau0 / Phi[n]
        Dt_info[n] = tau0 / (Phi[n] * (eps_info + H[n]))

    return {"Phi":Phi, "Dt":Dt, "Dt_info":Dt_info, "H":H}

        blocks.append(s)
